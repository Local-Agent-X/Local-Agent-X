# Local Agent X voice GPU sidecar — pinned dependencies.

# CUDA: faster-whisper bundles CTranslate2 with CUDA support.
# kokoro-onnx uses onnxruntime-gpu for CUDA. silero-vad uses torch (CUDA
# when available). All assume CUDA 12.x runtime libs from the wheels.

faster-whisper==1.1.0
# faster-whisper 1.1.0 (2024) predates huggingface_hub 1.x and does not pin it.
# hub 1.x dropped its `requests` dependency for httpx, which silently removed
# `requests` from the venv — and faster_whisper/utils.py imports requests at
# module scope without declaring it, so an unpinned hub resolves to 1.x and
# faster_whisper stops importing entirely. Cap the hub at the era this
# faster-whisper was built against, and declare requests explicitly rather than
# inheriting it by luck from whatever the hub happens to depend on.
huggingface_hub>=0.13,<1.0
requests>=2.31,<3.0
kokoro-onnx==0.4.9
# kokoro-onnx phonemizes via the `phonemizer-fork` distribution, NOT upstream
# `phonemizer` — both install the same `phonemizer` import package, so pinning
# the upstream name here both fails to resolve (upstream stops at 3.3.0) and,
# if it did resolve, would shadow the fork. This is the distribution
# _server/models.py patches for the words-mismatch guard; keep the pin here so
# that patch target stays at a known version.
phonemizer-fork==3.3.2
onnxruntime-gpu==1.20.1
silero-vad==5.1.2
soundfile==0.12.1
numpy>=2.0.2,<3.0  # floor is kokoro-onnx 0.4.9's requirement; Coqui/RVC's numpy<2 constraint lives in its own venv (see footer), not here
fastapi==0.115.6
uvicorn[standard]==0.32.1
websockets==14.1
torch==2.5.1
torchaudio==2.5.1
# CUDA runtime DLLs (cublas + cudnn) bundled as Python wheels — no need to
# install the full CUDA Toolkit. faster-whisper / CTranslate2 needs cublas64_12.dll
# at runtime; server.py prepends the wheel's bin dir to PATH on startup.
nvidia-cublas-cu12==12.4.5.8
nvidia-cudnn-cu12==9.1.0.70
