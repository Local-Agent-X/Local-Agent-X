    if (process.env.LAX_VOICE_DEBUG) {

        `[tier4/kokoro] ${activeDevice}+${activeDtype} init failed (${(e as Error).message}); falling back to cpu+q8`,
      );
    }
    activeDevice = "cpu";
    activeDtype = "q8";
